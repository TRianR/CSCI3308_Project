 18 var MY_SCORE = 0;

 26 /** ## Player Sprite */
 27 Q.Sprite.extend("Player",{
 28 
 29   /** the init constructor is called on creation */
 30   init: function(p) {
 31 
 32     /** You can call the parent's constructor with this._super(..) */
 33     this._super(p, {
 34       /**sprite "player_anim", */
 35       sheet: "player_anim", /** Setting a sprite sheet sets sprite width and height */
 36  	  x: 410,           
 37       y: 90             
 38     });
 39 
 40     this.add('2d, platformerControls, animation');
 41     this.on('walk');
 42     this.on('jump');
 43     
 44     this.on("hit.sprite",function(collision) {
 45 
 46       if(collision.obj.isA("Tower")) {
 47 		/** Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 48 		Q.audio.play('well.mp3');
 49        Q.stageScene("stageNextLevel", 1, {buttontext : "Oh me oh my", label: "Timmy fell down the well!" }); 
 50         this.destroy();
 51       }
 52       else if(collision.obj.isA("Ufo")) {
 53 	    /** Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 54 		Q.audio.play('well.mp3');
 55        Q.stageScene("stageNextLevel", 1, {buttontext : "Mom...MOM!!!", label: "Where to now?" }); 
 56         this.destroy();
 57       }
 58        else if(collision.obj.isA("Portal")) {
 59 	    /** Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 60 		Q.audio.play('well.mp3'); 
 61         Q.stageScene("stageNextLevel", 1, {buttontext : "Huzzah!", label: "On the the next dimension!!" }); 
 62         this.destroy();
 63       }
 64       else if(collision.obj.isA("Door")) {
 65 	    /** Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 66 		Q.audio.play('well.mp3');
 67 		Q.stageScene("stageNextLevel", 1, {buttontext : "Huh?", label: "Rubber Baby Buggy Bumper Rager!!!" });  
 68         this.destroy();
 69       }
 70 	  else if(collision.obj.isA("riceball")) {
 71 	    /**  Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 72 		Q.audio.play('well.mp3');
 73 		Q.stageScene("stageNextLevel", 1, {buttontext : "I think I have a stomach ache!", label: "Yummy" });  
 74         this.destroy();
 75       }
 76       else if(collision.obj.isA("green_Psyche")) {
 77 	    /** Timmy going down well sound -> 'well.mp3/.wav/.ogg' */
 78 		Q.audio.play('well.mp3');
 79 		Q.stageScene("stageNextLevel", 1, {buttontext : "Huweeeee", label: "When will it be over??" });  
 80         this.destroy();
 81       }
 82     });
 83   },
 84   
 85   walk: function(dt) {
 86     	if(this.p.vx != 0) { 
 87     		this.play("walk");
 88     	} else {
 89     		this.play("stand"); 
 90     	}
 91   	},
 92   	
 93   jump: function(dt) {
 94   	if(this.p.vy != 0) {
 95    		this.play("jump");
 96   	}
 97   }
 98

157 /** ## Enemy Sprite */
158 Q.Sprite.extend("Enemy",{
159   init: function(p) {
160     this._super(p, { sheet: 'enemy', vx: 100 });
161 
162     this.add('2d, aiBounce');
163 
164     /** Listen for a sprite collision, if it's the player,
165     // end the game unless the enemy is hit on top */
166     this.on("bump.left,bump.right,bump.bottom",function(collision) {
167       if(collision.obj.isA("Player")) { 
168 		  /** Play hurt audio */
169 		  Q.audio.play('Hit_Hurt.mp3');
170       	if(MY_LIVES <= 0) {
171 		/** Stops music and all sounds upon game ending */
172 		Q.audio.stop();
173 		MY_LEVEL = 1;
174       	Q.stageScene('hud', 3, { score: MY_SCORE, lives : MY_LIVES})
175         Q.stageScene("endGame",1, { label: "Poor little Timmy died!" }); 
176         collision.obj.destroy();
177         }
178         else { 
179         	MY_LIVES -= 1;
180         	Q.stageScene('hud', 3, { score: MY_SCORE, lives : MY_LIVES});
181         }
182       }
183     });
184 
185     /** If the enemy gets hit on the top, destroy it
186     // and give the user a "hop" */
187     this.on("bump.top",function(collision) {
188       if(collision.obj.isA("Player")) { 
189         this.destroy();
190         collision.obj.p.vy = -300;
191         MY_SCORE += 10
192         
193         /** play jump audio when landing ontop of enemy */
194         Q.audio.play('Jump.mp3');
195         
196        Q.stageScene('hud', 3, { score: MY_SCORE, lives : MY_LIVES});
197       }
198     });
199   }
